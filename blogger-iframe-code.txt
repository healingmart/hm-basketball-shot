<script>
(function(){
  var wrap=document.getElementById('hmBasketballFullscreenWrap');
  if(!wrap){return;}
  function mountBasketball(){
    if(window.innerWidth<=900){
      if(wrap.parentNode!==document.body){document.body.appendChild(wrap);}
      document.documentElement.classList.add('hm-basketball-page-lock');
    }else{
      document.documentElement.classList.remove('hm-basketball-page-lock');
    }
  }
  if(document.readyState==='loading'){
    document.addEventListener('DOMContentLoaded',mountBasketball,{once:true});
  }else{
    mountBasketball();
  }
  window.addEventListener('resize',mountBasketball,{passive:true});
  window.addEventListener('orientationchange',function(){setTimeout(mountBasketball,120);},{passive:true});
})();
</script>
